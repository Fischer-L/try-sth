let p = document.querySelector("p"); let inp = document.querySelector("input"); let btn = document.querySelector("button");
btn.addEventListener("mouseup", function () { p.innerText = "mouseup: Focus input after 3 secs"; setTimeout(function () { p.innerText = "mouseup: Focused input"; inp.focus(); }, 3000); });
btn.addEventListener("touchend", function () { p.innerText = "touchend: Focus input after 3 secs"; setTimeout(function () { p.innerText = "touchend: Focused input"; inp.focus(); }, 3000); });
